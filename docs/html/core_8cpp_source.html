93 void TrueNorthCore::forward_event(tw_bf *bf, nemo_message *m, tw_lp *lp) {
94  // Messages are core->core
95  // if this is a message from a core's output, integrate.
96  // if this message is a heartbeat from ourselves, then we will
97  // do leaks, get firing info, and do resets.
98  // we also need to set the current simulated neuromorphic "tick", and
99  // check to see if the message (if it is a spike) has arrived in an invalid order.
100  // EG: tick = 1, spike from neuron scheduled for 2, and heartbeat message is out!
101  // this should not happen.
102  evt_stat = BF_Event_Status ::None;
103  unsigned int random_call_count = lp->rng->count;
104  //check/update neuro tick
105  if (m->intended_neuro_tick > this->current_neuro_tick){
106  evt_stat |= BF_Event_Status ::NS_Tick_Update;
107  this->current_neuro_tick = get_neurosynaptic_tick(tw_now(lp));
108  }
109 
110  if(m->message_type == NEURON_SPIKE){
111  evt_stat |= BF_Event_Status ::Spike_Rec;
112  if ( ! this->heartbeat_sent){
113  // send the heartbeat message:
114  evt_stat |= BF_Event_Status::Heartbeat_Sent;
115  this->send_heartbeat(lp, bf);
116  this->heartbeat_sent = true;
117  }
118 
119  // integrate
120  this->integrate(lp, 0);
121  if(this->last_active_time < tw_now(lp)){
122  this->last_active_time = tw_now(lp);
123  }else{
124  tw_warning(TW_LOC, "Last active time was %d, current time is: %d. \n", this->last_active_time, tw_now(lp));
125  }
126  }else if(m->message_type == HEARTBEAT){
127  // heartbeat message rcvd. We need to leak, fire, reset.
128  //first though, set up the heartbeat state.
129  evt_stat |= BF_Event_Status :: Heartbeat_Rec;
130  this->heartbeat_sent = false;
131  this->leak(lp, bf);
132  if(this->last_leak_time < tw_now(lp)){
133  this->last_leak_time = tw_now(lp);
134  }else{
135  tw_warning(TW_LOC, "Last leak time was $d, current time is %d. \n", this->last_leak_time, tw_now(lp));
136  }
137  //ringing at 0 check
138  this->ringing();
139 
140  //bool did_fire = this->fire(lp, bf);
141  this->fire_reset(lp);
142  // Send those fire events:
143  for(int i = 0; i < NEURONS_PER_TN_CORE; i ++){
144  if(this->fire_status[i]){
145  evt_stat |= BF_Event_Status :: Spike_Sent;
146  // output neurons do not actually send spikes - we will collect them for output layers //
147  if( is_output_neuron(i)){
149  }else {
150 
151  for (int j = 0; j < MAX_OUTPUT_PER_TN_NEURON; j++) {
152  tw_lpid dest_gid = get_gid_from_core_local(destination_cores[i][j], destination_axons[i][j]);
153  tw_stime dest_time = this->delays[i] + JITTER(lp->rng, random_call_count);
154  tw_event *evt = tw_event_new(dest_gid, dest_time, lp);
155  nemo_message *msg = (nemo_message *) tw_event_data(evt);
156  msg->message_type = NEURON_SPIKE;

162  msg->debug_time = tw_now(lp);

164  }
165  }
166  }
167  }
168 
169  this->last_active_time = tw_now(lp);

295 void TrueNorthCore::fire_reset(tw_lp *lp){
296  //Handles fire and reset of neurons - post leak.
297  // good candidate for omp
298 
299  for (bool &fire_statu : fire_status) {
300  fire_statu = false;
301  }
302 #pragma omp parallel for
303  for(int neuron_id = 0; neuron_id < NEURONS_PER_TN_CORE; neuron_id ++){
304  // Combines fire and reset functions, since they are interlinked in the TN model
305  auto nj = (nemo_thresh_type) tw_rand_integer(lp->rng, 0, 255);
306 
307  fire_status[neuron_id] = membrane_potentials[neuron_id] >= positive_threshold[neuron_id] + nj;
308  // damn if statements!
309  if(fire_status[neuron_id]){
310  membrane_potentials[neuron_id] = (
311  (dt(reset_modes[neuron_id]) * reset_voltages[neuron_id]) + //δ(γj)Rj
312  (dt(reset_modes[neuron_id] - 1) * (membrane_potentials[neuron_id] - (positive_threshold[neuron_id] + nj))) + //δ(γj − 1)(Vj(t) − (αj + ηj))
313  (dt(reset_modes[neuron_id] - 2) * (membrane_potentials[neuron_id])) //δ(γj −2)Vj(t)
314  );
315  }else{ //j(t) < −[βjκj + (βj + ηj)(1 − κj)]
317  int neg_th = membrane_potentials[neuron_id] < (long) ( 0 - (negative_threshold[neuron_id] * kappa_vals[neuron_id] +
318  (negative_threshold[neuron_id] + nj) *(1-kappa_vals[neuron_id]))
319  );
320 
321  int rrv = (((-1*negative_threshold[neuron_id])*kappa_vals[neuron_id]) +
322  (((-1*(dt(reset_modes[neuron_id])))*reset_voltages[neuron_id]) +
323  ((dt((reset_modes[neuron_id] - 1)))*
324  (membrane_potentials[neuron_id]+ (negative_threshold[neuron_id] + (unsigned int) tw_rand_integer(lp->rng, 0, random_range_rst[neuron_id])))) +
325  ((dt((reset_modes[neuron_id] - 2)))*membrane_potentials[neuron_id]))*
326  (1 - kappa_vals[neuron_id]));
327 
328  membrane_potentials[neuron_id] = ( (neg_th * rrv) + (1 - neg_th) * membrane_potentials[neuron_id]);
329

376 void TrueNorthCore::leak(tw_lp *lp, tw_bf *bf) {
377 
378 #pragma omp parallel for
379  for(int neuron_id = 0; neuron_id < NEURONS_PER_TN_CORE; neuron_id ++) {
380  int num_missed_leaks = current_neuro_tick - (int)last_leak_time;
381  for(int i = 0; i < num_missed_leaks; i ++) {
383  nemo_weight_type pj = tw_rand_integer(lp->rng, 0, random_range_leak[neuron_id]);
384  nemo_weight_type epsilon_j = epsilons[neuron_id];
385  nemo_weight_type V_j = membrane_potentials[neuron_id];
386  short lambda = lambdas[neuron_id];
387  short c_jl = c_vals[neuron_id];
388 
389 
390  nemo_weight_type omega = (sigma_ls[neuron_id] * (1 - epsilon_j)) + (epsilon_j * sgn(V_j));
391 
392 
393  nemo_weight_type result = V_j + omega * (
394  (1 - c_jl) * lambda +
395  c_jl * bincomp(lambda, pj) * sgn(lambda)
396  );
397  // Ringing check:
398  if (epsilon_j == 1 && (sgn(result) != sgn(membrane_potentials[neuron_id]))) {
399  membrane_potentials[neuron_id] = 0;
400  } else {
401  membrane_potentials[neuron_id] += result;
